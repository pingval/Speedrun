(defvar ll '())

(defun scan ()
  (interactive)
  (setq al '())
  (goto-char 0)
  (while (scan-buffer "^\\(.\\).+$"
                      :regexp t
                      :tail t)
    (push (match-string 0)
          ll))
  (setq ll (reverse ll)))

(defun repl ()
  (interactive)
  (unwind-protect
      (while (scan-buffer "<dt>\\(.\\).*</dt>"
                          :regexp t
                          :tail t)
        (replace-match (concat "<dt>" (pop ll) "</dt>")))
    (setq ll '())))

(defun make-title-al (al)
  (mapcar #'(lambda (i)
              (let ((c (elt (car i) 0)))
                (cons c
                      ;; (format nil "~{~a ~}" (cdr i))
                      (let ((d (cdr i))
                            (str ""))
                        (mapcar #'(lambda (s)
                                    (dotimes (i (length s))
                                      (let ((sc (elt s i)))
                                        (if (char= sc c)
                                            (setq str (concat str (format nil "[~a]" sc)))
                                          (setq str (concat str (format nil "~a" sc))))))
                                    (setq str (concat str " ")))
                                d)
                        str))))
          al))

(defun print-alist (al)
  (goto-char (point-max))
  (dolist (i al)
    (let ((c (elt (car i) 0))
          (d (cdr i))
          (cnt (list 0 0 0 0))
          (tal (make-title-al al)))
      ;; (insert (format nil "~a:~{ ~a~}~%" c d))
      (insert "  <dt>" c "</dt>\n")
      (mapcar #'(lambda (s)
                  (insert "  <dd>")
                  (dotimes (i (length s))
                    (let ((sc (elt s i)))
                      (insert "<span title=\""
                              (cdr (assoc sc tal :test #'char=))
                              "\""
                              (format nil " class=\"num-~a ~a\""
                                      (if (<= (length (cdr (assoc (string sc) al :test #'string=))) 4)
                                          (length (cdr (assoc (string sc) al :test #'string=)))
                                        "many")
                                      (if (char= c sc) "current" ""))
                              ">"
                              sc
                              "</span>")))
                  (insert "</dd>\n"))
              d)
      )))

(defun f ()
  (interactive)
  (let ((l '()))
    (goto-char 0)
    (while (scan-buffer "."
                        :regexp t
                        :tail t)
      (let ((c (match-string 0)))
        (unless (assoc c l :test #'string=)
          (save-excursion
            (goto-char 0)
            (let ((re (concat "^.*" c ".*$"))
                  (d '()))
              (while (scan-buffer re
                                  :regexp t
                                  :tail t)
                (push (match-string 0) d))
              (push (cons c (reverse d)) l))))))
    (print-alist (sort (reverse l) #'< :key #'length))
    ))

(defun print-alist (al)
  (goto-char (point-max))
  (dolist (i al)
    (let ((c (elt (car i) 0))
          (d (cdr i))
          (cnt (list 0 0 0 0)))
      ;; (insert (format nil "~a:~{ ~a~}~%" c d))
      (insert c ":")
      (mapcar #'(lambda (s)
                  (insert " ")
                  (dotimes (i (length s))
                    (let ((sc (elt s i)))
                      (if (char= sc c)
                          (progn
                            (incf (nth i cnt))
                            (insert "[" sc "]"))
                        (insert sc)))))
              d)
      (when (>= (length d) 2)
        (if (member-if #'(lambda (n) (>= n 10)) cnt)
            (insert (format nil " ;;~{ ~d~}" cnt))
          (insert (format nil " ;; ~{~d~}" cnt))))
      (insert "\n")
      )))
